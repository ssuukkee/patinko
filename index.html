import random import string from flask import Flask, render_template, request app = Flask(__name__) def generate_password(length=12, use_digits=True, use_symbols=True): """安全なパスワードを生成する関数""" characters = string.ascii_letters if use_digits: characters += string.digits if use_symbols: characters += string.punctuation password = ''.join(random.choice(characters) for _ in range(length)) return password @app.route("/", methods=["GET", "POST"]) # ✅ ここでPOSTを許可 def index(): password = None if request.method == "POST": length = int(request.form.get("length", 12)) use_digits = "digits" in request.form use_symbols = "symbols" in request.form password = generate_password(length, use_digits, use_symbols) return render_template("index.html", password=password) if __name__ == "__main__": app.run(debug=True)
