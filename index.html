import random import string def generate_password(length=12, use_digits=True, use_special_chars=True): letters = string.ascii_letters # a-z, A-Z digits = string.digits if use_digits else '' # 0-9 special_chars = string.punctuation if use_special_chars else '' # !@#$... all_chars = letters + digits + special_chars if not all_chars: raise ValueError("At least one character set must be selected.") password = ''.join(random.choice(all_chars) for _ in range(length)) return password if __name__ == "__main__": length = int(input("Enter password length: ")) use_digits = input("Include numbers? (y/n): ").lower() == 'y' use_special_chars = input("Include special characters? (y/n): ").lower() == 'y' password = generate_password(length, use_digits, use_special_chars) print(f"Generated Password: {password}")
